/**
 * 解析出pgt
 * @param body
 * @return {String} pgtId
 */
function parseResponse(body) {
  var pgt = '';
  var result = body.match(/action="([\s\S]*?)"/);
  if (result) {
    result = result[1];
 
    var uri = url.parse(result, true);
    var pathname = uri.pathname;
 
    pgt = pathname.substr(pathname.lastIndexOf('/') + 1);
  }
 
  return pgt;
}
